Создание ролей

Группа без права певоначального подключения

create role myrole;

Группа с правами певоначального подключения

create role myrole with login;

Добавление дополнительных разрешений

grant connect, select on database my_database to my_role;

Включение пользователя в группу

grant my_role to my_user;

Исключение пользователя из группы

revoke my_role from my_user;
