Генерация значения для чисел numeric со значениями от и до

create or replace function random_between(min_val numeric, max_val numeric, round_to int=0) returns numeric as $$ declare value numeric = random() * (min_val - max_val) + max_val; begin if round_to = 0 then return floor(value); else return round(value, round_to); end if; end $$ language 'plpgsql';

Пример:

select time, device_id, random_between(3, 100, 4) as cpu, random_between(28, 83) as temperature_c from generate_series(now() - interval '1 hour', now(), interval '1 minute') as time, generate_series(1, 10, 1) as device_id;
