Файлы для запуска и остановки на windows

В составе установки Postgres на windows создаются дополнительные файлы для запуска и остановки службы, если случайно удалили или просто забыли как это выглядит...

start.bat

echo off sc start "postgresql-16" pause

stop.bat

echo off "D:\Program Files\PostgreSQL\16\bin\pg_ctl.exe" stop -D "D:\PGDATABASE" -m fast pause

runpgsql.bat

@echo off chcp 1251 PATH D:\Program Files\PostgreSQL\16\bin;%PATH% if not exist "%APPDATA%\postgresql" md "%APPDATA%\postgresql" psql.exe -h localhost -U "postgres" -d postgres -p 5433 pause

restart.bat

echo off "D:\Program Files\PostgreSQL\16\bin\pg_ctl.exe" stop -D "D:\PGDATABASE" -m fast sc start "postgresql-16" pause

reload.bat

echo off "D:\Program Files\PostgreSQL\16\bin\pg_ctl.exe" reload -D "D:\PGDATABASE" pause
