Псевдо колонка XMIN

XMIN - это псевдоколонка PostgreSQL, которая используется для отслеживания одновременного доступа к строкам. Используя XMIN, вы можете получить точное число, которое используется внутри операций MVCC.

create table tbl_MVCCDemo ( id int, code character varying );
insert into tbl_MVCCDemo values (1, 'XVK98');
select xmin, id, code from tbl_MVCCDemo;

update tbl_MVCCDemo set code = 'T2738' where XMIN = 1579;
select xmin, id, code from tbl_MVCCDemo;
